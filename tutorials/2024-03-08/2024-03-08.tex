                let $T_{i}(n) = i \cdot n$ and $f(n) = n$
		            \begin{align*}
			            g(n) & = \frac{n(n+1)}{2} \cdot f(n) \\
			                 & = O(n^2f(n)) \\
                       & = O(n^3)
			            .\end{align*}
                  \[
                    \therefore g(n) \neq O(f(n)) \text{and }\neq O(n(f(n))
                  .\] 

                note: this is because
                \[
                \sum^{n}_{i=1}  O(f(n)) \neq  O(\sum^{n}_{i=1} f(n))
                .\] 
	      \end{enumerate}
